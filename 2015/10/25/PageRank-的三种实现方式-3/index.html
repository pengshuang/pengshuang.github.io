PageRank-的三种实现方式(3)

发表于 2015-10-25   |   分类于 Hadoop   |  

这篇文章我们来介绍一下用并行的思想来实现PageRank，分别用Hadoop和Spark来实现。

之前我们介绍了Google的PageRank算法模型，实现了对全互联网网页的打分。也实现一个Java版单机的PageRank算法，但是，我们需要注意的是，对于互联网上海量数据的处理，在单机下是不可能实现的，所以如何将PageRank并行计算，将是本文的重点。

首先，我们来介绍一下Hadoop的MapReduce计算框架实现方法，和以前一样，讲解我会穿插在代码注释中。

首先是Map类：

package org.tseg.pagerank;

import java.io.IOException;
import java.util.StringTokenizer;

import org.apache.hadoop.io.IntWritable;
import org.apache.hadoop.io.Text;
import org.apache.hadoop.mapreduce.Mapper;

public class MyMapper extends Mapper<Object, Text, IntWritable, Text> {

//IntWritable 即是Hadoop里面的int，其他类似
private IntWritable id;
private String pr;
private int count;
private float average_pr;

public void map(Object key, Text value, Context context) throws IOException, InterruptedException {

    //StringTokenizer方法自动以空格为间隔，返回值为Enumeration<Object>类型的一个容器，容器中的每个元素为之前由空格分开后的元素组成
    StringTokenizer str = new StringTokenizer(value.toString());
    //遍历整个容器，第一步取id为测试网页的序号
    if(str.hasMoreTokens()) {
        id = new IntWritable(Integer.parseInt(str.nextToken()));
    }else{
        return;
    }
    //pr代表测试网页的初始权值
    pr = str.nextToken();
    //count代表每个网页链接其他网页的个数
    count = str.countTokens();
    //average_pr 为平均贡献值
    average_pr = Float.parseFloat(pr) / count ;
    while(str.hasMoreTokens()) {
        String nextId = str.nextToken();
        IntWritable next_id = new IntWritable(Integer.parseInt(nextId));
        //Map函数输出给Reduce函数的类型有2种，第一种为（链接网页序号，“@”+对链接网页的平均贡献值）
        Text t = new Text("@"+average_pr);
        context.write(next_id, t);
        //第二种为（网页序号，"&"+它所链接的网页）
        Text tt = new Text("&"+nextId);
        context.write(id,tt);
        }

    }

}

然后是我们的Reduce类，Reduce类简单一些，

package org.tseg.pagerank;

import java.io.IOException;
import java.util.ArrayList;

import org.apache.hadoop.io.FloatWritable;
import org.apache.hadoop.io.IntWritable;
import org.apache.hadoop.io.Text;
import org.apache.hadoop.mapreduce.Reducer;

public class MyReducer extends Reducer<IntWritable, Text, IntWritable, Text> {

public void reduce(IntWritable key, Iterable<Text> values, Context context) throws IOException, InterruptedException {

    ArrayList<String> idlist = new ArrayList<String>();
    String link = " ";
    float pr = 0;
    //提取贡献值，并累加起来，用pr保存
    for(Text id : values) {
        String id2 = id.toString();
        if(id2.substring(0, 1).equals("@")){
            pr += Float.parseFloat(id2.substring(1));
        }
        //把所有链接的节点存放于idlist中
        else if (id2.substring(0,1).equals("&")) {
            idlist.add(nextid);
        }
    }

    pr = pr * 0.85f + 0.15f;
    for(int i = 0; i< idlist.size();i++){
        link += idlist.get(i) + " ";
    }

    String result = String.valueOf(pr);
    System.out.println("Reduce  result= " + result);
    context.write(key, new Text(result));
    }
}

package org.tseg.pagerank;

import java.io.IOException;

import org.apache.hadoop.conf.Configuration;
import org.apache.hadoop.fs.Path;
import org.apache.hadoop.io.IntWritable;
import org.apache.hadoop.io.Text;
import org.apache.hadoop.mapreduce.Job;
import org.apache.hadoop.mapreduce.lib.input.FileInputFormat;
import org.apache.hadoop.mapreduce.lib.output.FileOutputFormat;

public class rank {
    public static void main(String[] args) throws IOException,
    InterruptedException, ClassNotFoundException {

    Configuration conf = new Configuration();
    String pathIn = "/Users/pengshuang/Desktop/pagerank/input";//输入路径  
    String pathOut ="/Users/pengshuang/Desktop/pagerank/1";//输出路径  
    //迭代10次
        Job job = Job.getInstance(conf, "rank");
        job.setJarByClass(rank.class);
        job.setMapperClass(MyMapper.class);
        job.setReducerClass(MyReducer.class);
        job.setOutputKeyClass(IntWritable.class);
        job.setOutputValueClass(Text.class);
        FileInputFormat.addInputPath(job, new Path(pathIn));
        FileOutputFormat.setOutputPath(job, new Path(pathOut));
        job.waitForCompletion(true);
    }
}

package spark

import org.apache.spark.{SparkConf, SparkContext}


object SparkPageRank {

  def main(args: Array[String]) {

      //设置测试环境为本地测试
    val sparkConf = new SparkConf().setAppName("PageRank").setMaster("local")
    val iters = if (args.length > 1) args(1).toInt else 10
    val ctx = new SparkContext(sparkConf)
    //读取测试文件
    val lines = ctx.textFile("file:///Users/pengshuang/Desktop/pagerank/input/twitter_combined.txt", 1)
    //每行用空格隔开
    val links = lines.map { s =>
      val parts = s.split("\\s+")
      (parts(0), parts(1))
    }.distinct().groupByKey().cache()
    var ranks = links.mapValues(v => 1.0)

for (i <- 1 to iters) {
  val contribs = links.join(ranks).values.flatMap { case (urls, rank) =>
    val size = urls.size
    urls.map(url => (url, rank / size))
  }
  ranks = contribs.reduceByKey(_ + _).mapValues(0.15 + 0.85 * _)
}
ranks.map(x => (x._2,x._1)).sortByKey(false).map(x => (x._2,x._1)).saveAsTextFile("file:///Users/pengshuang/Desktop/pagerank/output")
ctx.stop()
    }
}
